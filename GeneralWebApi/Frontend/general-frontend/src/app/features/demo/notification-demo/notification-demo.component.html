üíª Usage Examples:

Basic Usage:

// Success notification
this.notificationService.success('Success', 'Operation completed!');

// Error notification  
this.notificationService.error('Error', 'Something went wrong!');

// Warning notification
this.notificationService.warning('Warning', 'Please check your input!');

// Info notification
this.notificationService.info('Info', 'Here is some information!');

API Error Handling:

// In HTTP interceptor
catchError((error: HttpErrorResponse) => {
  switch (error.status) {
    case 400:
      this.notificationService.error('Bad Request', 'Invalid data');
      break;
    case 401:
      this.notificationService.warning('Unauthorized', 'Please login');
      break;
    case 500:
      this.notificationService.error('Server Error', 'Try again later');
      break;
  }
  return throwError(error);
})

Loading States:

// Show loading notification
const loadingId = this.notificationService.show({
  title: 'Loading...',
  message: 'Please wait...',
  type: 'info',
  duration: 0,
  persistent: true
});

// Hide loading and show result
this.notificationService.remove(loadingId);
this.notificationService.success('Complete', 'Done!');
